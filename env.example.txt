# ============ OpenAI Configuration ============
# Get your API key from: https://platform.openai.com/api-keys
OPENAI_API_KEY=sk-your-openai-api-key-here

# ============ Server Configuration ============
# Port for the relay server (default: 8081)
PORT=8081

# Relay server URL (for React app to connect to)
REACT_APP_LOCAL_RELAY_SERVER_URL=http://localhost:8081

# ============ Firebase Client Configuration ============
# Get these from Firebase Console > Project Settings > General > Your apps
REACT_APP_FIREBASE_API_KEY=AIza...
REACT_APP_FIREBASE_AUTH_DOMAIN=your-project.firebaseapp.com
REACT_APP_FIREBASE_PROJECT_ID=your-project-id
REACT_APP_FIREBASE_STORAGE_BUCKET=your-project.appspot.com
REACT_APP_FIREBASE_MESSAGING_SENDER_ID=123456789
REACT_APP_FIREBASE_APP_ID=1:123456789:web:abc123

# ============ Firebase Admin Configuration ============
# Download service account key from:
# Firebase Console > Project Settings > Service Accounts > Generate New Private Key
# Paste the ENTIRE JSON content as a single line string
FIREBASE_SERVICE_ACCOUNT_KEY={"type":"service_account","project_id":"your-project","private_key_id":"...","private_key":"-----BEGIN PRIVATE KEY-----\n...\n-----END PRIVATE KEY-----\n","client_email":"firebase-adminsdk-xxx@your-project.iam.gserviceaccount.com","client_id":"..."}

# ============ Optional: Development Settings ============
# Set to 'development' for verbose logging
NODE_ENV=production

# ============ Notes ============
# - Keep your .env file secret!
# - Don't share your API keys
# - Add .env to .gitignore
